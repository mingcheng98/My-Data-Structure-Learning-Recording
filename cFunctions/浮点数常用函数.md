# 浮点数常用函数

## 1.浮点数的大小比较及其与整数的关系

```c
#define eps 1e-9
//judge whether x is an integer
int intJudge(double x){
    if(fabs(x - round(x)) < eps) return 1;
    else return 0;
}

//judge whether x and y are equal
int equalJudge(double x, double y){
    if(fabs(x - y) < eps) return 1;
    else return 0;
}

int main(){
    double test;
    scanf("%lf", &test);
    printf("%lf\n",  fabs(test)); // get the absolute value
    printf("%lf\n", floor(test)); // down to nearest integer
    printf("%lf\n",  ceil(test)); //   up to nearest integer
    printf("%lf\n", round(test)); //round to nearest integer
    printf("%lf\n", trunc(test)); //round towards zero to nearest integer
    return 0;
}
```

## 2.一元二次方程求根

```c
#define eps 1e-9
void solveEquation(double a, double b, double c){
    if(fabs(a) > eps){
        double delta = b * b - 4 * a * c;
        if(fabs(delta) < eps) printf("Same two roots: %lf\n", -b / (2 * a));
        else if(delta > 0){
            double x1 = (-b + sqrt(delta)) / (2 * a);
            double x2 = (-b - sqrt(delta)) / (2 * a);
            printf("Two different roots: %lf and %lf\n", x1, x2);
        }else{
            printf("No real root\n");
        }
    }else if(fabs(b) > eps){
        //a == 0 && b != 0
        printf("One root: %lf\n", -c / b);
    }else{
        //a == 0 && b == 0
        if(fabs(c) < eps) printf("Infinite solutions\n");
        else printf("No solution\n");
    }
    return;
}
```
