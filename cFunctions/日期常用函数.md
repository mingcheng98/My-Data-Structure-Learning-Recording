# 日期常用函数

## 1.判断某一年是否是闰年

```c
int isLeap(int year){
	return ((year % 4 == 0 && year %100 != 0) || year % 400 == 0);
}
```

## 2.判断某一天是星期几

```c
int getWeek(int year, int month, int day){
	int century, weekday;
	if(month < 3){
		year--;
		month += 12;
	}
	century = year / 100;
	year %= 100;
	//Zeller Formula
	weekday = (year + year/4 + century/4 - 2*century + (26*(month+1))/10 + day - 1) % 7;
	while(weekday <= 0){
		weekday += 7;
	}
	return weekday;
}
```

## 3.计算某两个日期相差的天数

```c
int deltaDate(int y1, int m1, int d1, int y2, int m2, int d2){
    int months[2][13] = {
        {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
        {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}
    };
    int sum = 0;
    if(y1 == y2){
        if(m1 == m2){
            sum += d2 - d1;
        }else{
            sum += months[isleap(y1)][m1]- d1;
            for(int i = m1 + 1; i < m2; ++i){
                sum += months[isLeap(y1)][i];
            }
            sum += d2;
        }
    }else{
        sum += months[isLeap(y1)][m1] - d1;
        for(int i = m1 + 1; i <= 12; ++i){
            sum += months[isLeap(y1)][i];
        }
        for(int i = y1 + 1; i < y2; ++i){
            sum += 365 + isLeap(i);
        }
        for(int i = 1; i < m2; ++i){
            sum += months[isLeap(y2)][i];
        }
        sum += d2;
    }
    return sum;
}
```

## 4.计算某个日期是该年的第几天

```c
int getDayth(int year, int month, int day){
    int months[2][13] = {
        {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
        {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}
    };
    int sum = 0;
    for(int i = 1; i < month; ++i){
        sum += months[isLeap(year)][i];
    }
    sum += day;
    return sum;
}
```

## 5.日期乱码转换为标准日期

```c
int trueDate(int year, int month, int day){
    int months[2][13] = {
        {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
        {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}
    };
    year += (month - 1) / 12; //legal year
    month = (month - 1) % 12 + 1; //legal month
    while(day > months[isLeap(year)][month]){
        //illegal day
        day -= months[isLeap(year)][month];
        month++;
        if(month > 12){
            month = 1;
            year++;
        }
    }
    return year * 10000 + month * 100 + day;
}
```
