# 数论中的常用函数

## 1.求两个整数的最大公约数（辗转相除法）

```c
int gcd(int a, int b){
	int r;
	if(b == 0) return a < 0 ? -a : a;
	while((r = a % b) != 0){
		a = b;
		b = r;  // gcd(a, b) == gcd(b, a%b)  (b != 0)
	}
	return b < 0 ? -b : b;
}
```

## 2.判断一个正整数是否是素数（提供欧拉筛思路）

```c
int isPrime(int x){
	if(x == 2 || x == 3 || x == 5 || x == 7){
		return 1;    //judge the first four primes
	}else if(x % 2 == 0 || x % 3 == 0 || x % 5 == 0 || x % 7 == 0 || x == 1){
		return 0;    //these numbers cannot be primes
	}else{
		int step = 4;
		for(int i = 7 + step; i * i <= x; i += step){
			//step can be exchanged between 2 and 4, primes mod 6 == 1 or 5
			if(x % i == 0) return 0;
			step = 6 - step;
		}   //prime divisor must less than sqrt(number)
	}
	return 1;
}
```

```c
#define MAXN 1000000
int isPrime[MAXN]; //1 means i is prime, 0 means not prime
int prime[MAXN];   //prime array to store all primes up to n
int primeCnt = 0;

//eulerSieve: mark all primes up to n, and store them in prime array
void eulerSieve(int n){
	for(int i = 0; i <= n; ++i){
		isPrime[i] = 1; //initialize all numbers as prime
	}
	isPrime[0] = isPrime[1] = 0; //0 and 1 are not prime
	for(int i = 2; i <= n; ++i){
		if(isPrime[i]) prime[primeCnt++] = i; //i is a prime, store it
		for(int j = 0; j < primeCnt && i * prime[j] <= n; ++j){
			isPrime[i * prime[j]] = 0; //mark "i * prime[j]" as not prime
			if(i % prime[j] == 0) break;
			//guarantee "i * prime[j]" marked as not prime only once,
            //which is the key to achieve O(n) time complexity
		}
	}
}
```

### 5.快速幂取模

```c
long long qpow(long long a, long long b, int p){
	long long ans = 1;
	a = a % p;
	while(b){
		if(b & 1) ans = (ans * a) % p;
		b >>= 1;
		a = a * a % p;
	}
	return ans;
}
```

### 6.大数积取模

```c
long long qmul(long long a, long long b, int mod){
	long long ans = 0;
	while(b){
		if(b & 1) ans = (ans + a) % mod;
		a = (a << 1) % mod;
		b >>= 1;
	}
	return ans;
}
```
